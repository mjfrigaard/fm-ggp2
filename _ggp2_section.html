# labels
labs_graph <- ggplot2::labs(title = "Title", 
                            subtitle = "subtitle", 
                            x = "X", y = "Y")
# layers
ggp2_graph <- ggplot2::ggplot(data = df, 
    mapping = aes(x = X, y = Y)) + 
    ggplot2::geom_blank()
# graph
ggp2_graph + 
    labs_graph
